25-Mile Marathon

Mary wants to run a 25-mile marathon. When she attempts to sign up for the marathon, she notices the sign-up sheet doesn't directly state the marathon's length. Instead, the marathon's length is listed in small, different portions. Help Mary find out how long the marathon actually is. Return true if the marathon is 25 miles long, otherwise, return false.

Solution :

function marathonDistance(d) { let total = 0; for (let i = 0; i < d.length; i++) { if (d[i] < 0) { total += -d[i]; } else { total += d[i]; } } return total === 25; }

How Much is True?

Create a function which returns the number of true values there are in an array...

Solution :

function countTrue(arr) { let count = 0; for (let i = 0; i < arr.length; i++) { if (arr[i] === true) { count++; } } return count; }

Square Every Digit

Create a function that squares every digit of a number.

Solution :

function squareDigits(n) { let str = String(n); let result = ""; for (let i = 0; i < str.length; i++) { let digit = Number(str[i]); result += (digit * digit); } return Number(result); }

Array of Multiples

Create a function that takes two numbers as arguments (num, length) and returns an array of multiples of num until the array length reaches length.

Solution :

function arrayOfMultiples(num, length) { let result = []; for (let i = 1; i <= length; i++) { result.push(num * i); } return result; }

Write a function that converts an object into an array of keys and values.

Special Arrays

An array is special if every even index contains an even number and every odd index contains an odd number. Create a function that returns true if an array is special, and false otherwise.

Solution :

function isSpecialArray(arr) { for (let i = 0; i < arr.length; i++) { if (i % 2 === 0 && arr[i] % 2 !== 0) { return false; } if (i % 2 !== 0 && arr[i] % 2 === 0) { return false; } } return true; }

Sum of Cubes

Create a function that takes in an array of numbers and returns the sum of its cubes.

Solution :

function sumOfCubes(nums) { if (!Array.isArray(nums) || nums.length === 0) { return 0; } let sum = 0; for (let i = 0; i < nums.length; i++) { let cube = nums[i] * nums[i] * nums[i]; sum += cube; } return sum; }

Is the Number Symmetrical?

Create a function that takes a number as an argument and returns true or false depending on whether the number is symmetrical or not. A number is symmetrical when it is the same as its reverse.

Solution :

function isSymmetrical(num) { let str = String(num); let len = str.length; for (let i = 0; i < len / 2; i++) { if (str[i] !== str[len - 1 - i]) { return false; } } return true; }

Solution :

Create a function that takes an array of numbers and return both the minimum and maximum numbers, in that order.

function findMinAndMax(numbers) { if (!Array.isArray(numbers) || numbers.length === 0) { throw new Error("Input must be a non-empty array of numbers."); } let min = numbers[0]; let max = numbers[0]; for (let i = 1; i < numbers.length; i++) { if (numbers[i] < min) { min = numbers[i]; } if (numbers[i] > max) { max = numbers[i]; } } return { min, max }; }
